WITH t AS (
  SELECT symbol, side, qty, price,
         CASE WHEN ts>1000000000000 THEN ts/1000 ELSE ts END AS ts_s
  FROM trades
)
SELECT symbol, side, qty, price,
       datetime(ts_s,'unixepoch','localtime') AS ts
FROM t
WHERE CAST(ts_s AS INTEGER) >= CAST(strftime('%s','now','-24 hours') AS INTEGER)
ORDER BY ts_s DESC
LIMIT 50;
"
